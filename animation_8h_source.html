123template <std::size_t ValueCount> class Animation : public AnimationType
124{
125public:
126 using ValueList = std::array<float, ValueCount>;
127 using SourceList = std::array<std::unique_ptr<AnimatedValue>, ValueCount>;
128 using UpdateFn = std::function<void (int, const ValueList&)>;
135 using CompletionFn = std::function<void (int, bool)>;
136
145 Animation (int id = 0)
146 : AnimationType { id }
147 {
148 }
149
156 Animation (SourceList&& sources, int id = 0)
157 : AnimationType { id }
158 , sources { std::move (sources) }
159 {
160 }
161
168 bool setValue (size_t index, std::unique_ptr<AnimatedValue> value)
169 {
170 if (index >= ValueCount)
171 {
172 jassertfalse;
173 return false;
174 }
175
176 sources[index] = std::move (value);
177 return true;
178 }
179
187 AnimatedValue* getValue (size_t index) override
188 {
189 if (index < ValueCount)
190 return sources[index].get ();
191
192 jassertfalse;
193 return nullptr;
194 }
195
201 void onUpdate (UpdateFn update) { updateFn = update; }
202
209 void onCompletion (CompletionFn complete) { completionFn = complete; }
210
218 Status gotoTime (juce::int64 timeInMs) override
219 {
220 if (finished)
221 {
222 if (completionFn != nullptr)
223 completionFn (getId (), false);
224 return Status::finished;
225 }
226
227 juce::int64 deltaTime;
228 // if this is the first time we're being executed, perform some setup:
229 if (startTime < 0)
230 {
231 startTime = lastTime = timeInMs;
232 deltaTime = 0;
233 }
234 else
235 {
236 deltaTime = timeInMs - lastTime;
237 lastTime = timeInMs;
238 }
239
240 const auto totalElapsed { timeInMs - startTime };
241
242 // if we're still delaying, just return.
243 if (totalElapsed < preDelay)
244 return Status::processing;
245
246 // recalculate the elapsed and delta times to account for an
247 // expired delay
248 const auto effectElapsed { totalElapsed - preDelay };
249 deltaTime = std::min (deltaTime, totalElapsed);
250
251 // loop through our value generators and update:
252 ValueList values;
253 int completeCount { 0 };
254
255 for (int i = 0; i < ValueCount; ++i)
256 {
257 auto& val = sources[i];
258 if (val != nullptr)
259 {
260 // values[i] = val->GetNextValue ();
261 values[i] = val->getNextValue (static_cast<int> (effectElapsed),
262 static_cast<int> (deltaTime));
263 completeCount += (val->isFinished ()) ? 1 : 0;
264 }
265 else
266 jassertfalse;
267 }
268
269 if (updateFn != nullptr)
270 updateFn (getId (), values);
271
272 if (completeCount == ValueCount)
273 finished = true;
274
275 return Status::processing;
276 }
277
278 void cancel (bool moveToEndPosition) override
279 {
280 for (auto& val : sources)
281 {
282 if (val != nullptr)
283 val->cancel (moveToEndPosition);
284 }
285
286 if (moveToEndPosition && updateFn != nullptr)
287 {
288 // send one more update where all of the individual values
289 // have snapped to their end states.
290 ValueList values;
291 for (int i = 0; i < ValueCount; ++i)
292 {
293 auto& val = sources[i];
294 jassert (val != nullptr);
295 if (val != nullptr)
296 values[i] = val->getEndValue ();
297 }
298 updateFn (getId (), values);
299 }
300
301 // notify that the effect is complete.
302 if (completionFn != nullptr)
303 completionFn (getId (), true);
304 finished = true;
305 }
306
307 bool isFinished () override { return finished; }
308
309 bool isReady () const override
310 {
311 for (auto& src : sources)
312 {
313 if (nullptr == src.get ())
314 return false;
315 }
316 return true;
317 }
318
319public:
322 UpdateFn updateFn;
323
325 CompletionFn completionFn;
326
327private:
329 juce::int64 startTime { -1 };
331 juce::int64 lastTime { -1 };
332
334 bool finished { false };
335
337 SourceList sources;
338};
