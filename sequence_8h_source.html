36template <int valueCount> class Sequence : public Animation<valueCount>
37{
38public:
39 Sequence (int id = 0)
40 : Animation<valueCount> (id)
41 , currentEffect { 0 }
42 {
43 }
44
45 bool isFinished () override { return (currentEffect >= sequence.size ()); }
46
47 bool isReady () const override
48 {
49 for (const auto& effect : sequence)
50 {
51 if ((nullptr == effect) || (!effect->isReady ()))
52 return false;
53 }
54 return true;
55 }
56
57 AnimationType::Status gotoTime (juce::int64 timeInMs) override
58 {
59 auto effect = getEffect (currentEffect);
60 if (effect)
61 {
62 if (AnimationType::Status::finished == effect->gotoTime (timeInMs))
63 ++currentEffect;
64
65 return isFinished () ? AnimationType::Status::finished
66 : AnimationType::Status::processing;
67 }
68
69 return AnimationType::Status::finished;
70 }
71
72 void cancel (bool moveToEndPosition) override
73 {
74 if (moveToEndPosition)
75 {
76 auto lastIndex = static_cast<int> (sequence.size () - 1);
77 auto lastEffect = getEffect (lastIndex);
78 if (lastEffect)
79 lastEffect->cancel (moveToEndPosition);
80 }
81 }
82
83 void addAnimation (std::unique_ptr<Animation<valueCount>> effect)
84 {
85 // We need to make each of the effects notify us on update or completion,
86 // so that we can pass those along to whoever passed in a single
87 // lambda to us.
88 effect->onUpdate (
89 [this] (int /*id*/, const typename Animation<valueCount>::ValueList& val)
90 {
91 if (this->updateFn != nullptr)
92 this->updateFn (this->getId (), val);
93 });
94
95 effect->onCompletion (
96 [this] (int /*id*/, bool wasCanceled)
97 {
98 // Each effect in the sequence will notify us, but we only pass
99 // along the final one.
100 if ((currentEffect == sequence.size () - 1) &&
101 this->completionFn != nullptr)
102 this->completionFn (this->getId (), wasCanceled);
103 });
104
105 sequence.push_back (std::move (effect));
106 }
107
108private:
114 Animation<valueCount>* getEffect (int index)
115 {
116 if (juce::isPositiveAndBelow (index, sequence.size ()))
117 return sequence[index].get ();
118
119 jassertfalse;
120 return nullptr;
121 }
122
123private:
124 std::vector<std::unique_ptr<Animation<valueCount>>> sequence;
125
126 int currentEffect { 0 };
127};
